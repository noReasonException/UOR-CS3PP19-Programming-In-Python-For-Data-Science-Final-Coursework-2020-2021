"""
We can also group our day into useful quantiles, extracting more useful info. let
0-6 : Early Morning
6-9: Morning Rush Hour
9-12: Morning
12-18: Afternoon
18-20 : Evening Rush Hour
20-23: Night
"""

def hour_day_group(hour):
    if(0<=hour<=6):return 0
